## Summary

Implement a Bootstrap 5 modal form for creating and editing courses. The modal handles multipart/form-data submissions (text fields + optional image), performs client-side validation with inline error messages, generates a live image preview using FileReader, and synchronizes state back to the parent `CourseViewComponent` in-place (no page reload). Department context is user-selected on Create and read from the course object on Edit.

```typescript
/**
 * Create a new course under a department.
 * Permission: Course:add
 * POST /api/Course/{departmentId}
 * ⚠️ Do NOT set Content-Type — browser sets multipart/form-data + boundary automatically.
 */
createCourse(departmentId: number, formData: FormData): Observable<Course> {
  return this.http.post<any>(`${this.baseUrl}/${departmentId}`, formData).pipe(
    map(c => this.normalizeCourse(c))
  );
}
```

#### A3 — Add `updateCourse()` to `CourseService`

```typescript
/**
 * Update an existing course.
 * Permission: Course:update
 * PUT /api/Course/{departmentId}/{id}
 * ⚠️ Do NOT set Content-Type manually.
 */
updateCourse(departmentId: number, courseId: number, formData: FormData): Observable<Course> {
  return this.http.put<any>(`${this.baseUrl}/${departmentId}/${courseId}`, formData).pipe(
    map(c => this.normalizeCourse(c))
  );
}
```

#### B3 — Form Initialization & Patch

```typescript
ngOnInit(): void {
  this.loadDepartments();
  this.initForm();
  if (this.courseData) {
    this.patchForm(this.courseData);
  }
}

initForm(): void {
  this.form = this.fb.group({
    title: ['', [Validators.required, Validators.minLength(5)]],
    description: ['', Validators.required],
    semster: [null, Validators.required],         // integer: 1, 2, 3
    academicLevel: [null, Validators.required],   // integer: 1–5
    credit_Hour: [null, [Validators.required, Validators.min(1), Validators.max(10)]],
    learningOutcomes: ['', Validators.required],
    departmentId: [null, Validators.required],
  });
}

patchForm(course: Course): void {
  this.form.patchValue({
    title: course.Title,
    description: course.Description,
    semster: course.semster,       // map string "Fall" to int 1 if needed
    academicLevel: course.academicLevel,
    credit_Hour: course.Credit_Hour,
    learningOutcomes: course.LearningOutcomes,
    departmentId: course.departmentId,
  });
  if (course.ImageUrl) {
    this.imagePreviewUrl = course.ImageUrl;
  }
}
```

**Note on `semster` mapping**: The backend returns a string ("Fall") from GET but expects an integer (1) on POST/PUT. The `patchForm` must convert: `{ Fall: 1, Spring: 2, Summer: 3 }`.

### Phase C — Image Handling

#### C1 — File Selection Handler

```typescript
onFileSelected(event: Event): void {
  const input = event.target as HTMLInputElement;
  if (!input.files?.length) return;

  const file = input.files[0];
  const maxSize = 5 * 1024 * 1024; // 5 MB
  const allowedTypes = ['image/jpeg', 'image/png', 'image/webp'];

  if (file.size > maxSize) {
    Swal.fire({ icon: 'error', title: 'File too large', text: 'Image must be under 5 MB.' });
    return;
  }
  if (!allowedTypes.includes(file.type)) {
    Swal.fire({ icon: 'error', title: 'Invalid format', text: 'Only JPEG, PNG, and WebP are allowed.' });
    return;
  }

  this.selectedFile = file;
  const reader = new FileReader();
  reader.onload = () => { this.imagePreviewUrl = reader.result as string; };
  reader.readAsDataURL(file);
}
```

### Phase D — Form Submission

#### D1 — Build FormData

```typescript
buildFormData(): FormData {
  const fd = new FormData();
  const v = this.form.value;
  fd.append('title', v.title);
  fd.append('description', v.description);
  fd.append('semster', String(v.semster));
  fd.append('academicLevel', String(v.academicLevel));
  fd.append('credit_Hour', String(v.credit_Hour));
  fd.append('learningOutcomes', v.learningOutcomes);
  if (this.selectedFile) {
    fd.append('ImageFile', this.selectedFile, this.selectedFile.name);
  }
  return fd;
}
```

#### D2 — Submit Handler

```typescript
onSubmit(): void {
  if (this.form.invalid) {
    this.form.markAllAsTouched();
    return;
  }

  this.isSaving = true;
  const fd = this.buildFormData();
  const departmentId = this.form.value.departmentId;

  const request$ = this.courseData
    ? this.courseService.updateCourse(departmentId, this.courseData.Id, fd)
    : this.courseService.createCourse(departmentId, fd);

  request$.subscribe({
    next: (savedCourse) => {
      this.isSaving = false;
      Swal.fire({ icon: 'success', title: this.courseData ? 'Updated!' : 'Created!', timer: 2000, showConfirmButton: false });
      if (this.courseData) {
        this.courseUpdated.emit(savedCourse);
      } else {
        this.courseCreated.emit(savedCourse);
      }
      this.closeModal();
    },
    error: (err) => {
      this.isSaving = false;
      const msg = err.error?.message || err.error || 'An unexpected error occurred.';
      Swal.fire({ icon: 'error', title: 'Save Failed', text: msg });
    }
  });
}
```

```typescript
// In-place state: Create
onCourseCreated(newCourse: Course): void {
  this.searchTerm = '';
  this.selectedSemester = '';
  this.courses.unshift(newCourse);
  this.applyFilters();
}

// In-place state: Edit
onCourseUpdated(updatedCourse: Course): void {
  const idx = this.courses.findIndex(c => c.Id === updatedCourse.Id);
  if (idx !== -1) {
    this.courses[idx] = updatedCourse;
    this.applyFilters();
  }
}
```
